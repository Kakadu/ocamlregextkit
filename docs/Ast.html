module Ast: sig .. end

Representation of a Regular Expression as an Abstract Syntax Tree

type re =
| Literal of string
| Epsilon
| Empty
| Union of re * re
| Concat of re * re
| Star of re 
val simplify : re -> re

simplify r attempts to simplify RE r by Kozen Axioms

val print : re -> unit

print r prints a string representation of the Abstract Syntax Tree representing r
